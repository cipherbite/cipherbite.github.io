Code Examples

[1]: Image GPS Extraction & Mapping 
from forensic_harvester import ImageAnalyzer
import folium

# Initialize analyzer
analyzer = ImageAnalyzer()

# Extract all metadata from suspicious image
metadata = analyzer.extract_metadata('suspect_photo.jpg')

# Extract GPS coordinates if present
if gps_data := metadata.get('gps_info'):
    lat, lon = gps_data['latitude'], gps_data['longitude']
    
    # Create interactive map
    map_obj = folium.Map(location=[lat, lon], zoom_start=15)
    folium.Marker([lat, lon], 
                  popup=f"Photo taken here
Time: {metadata['timestamp']}").add_to(map_obj)
    
    print(f"üìç Location found: {lat:.6f}, {lon:.6f}")
    print(f"üìÖ Timestamp: {metadata['timestamp']}")
    print(f"üì± Device: {metadata['camera_model']}")
