[1]: Entropy-based Detection Algorithm 
import math
from collections import Counter

class DNSExfiltrationDetector:
    def __init__(self, threshold=3.5):
        self.entropy_threshold = threshold
        self.baseline_domains = self.load_baseline()
        
    def calculate_entropy(self, domain):
        """Calculate Shannon entropy of domain name"""
        # Remove TLD for analysis
        subdomain = domain.split('.')[0]
        
        # Calculate character frequency
        freq = Counter(subdomain)
        length = len(subdomain)
        
        # Shannon entropy calculation
        entropy = -sum(
            (count/length) * math.log2(count/length) 
            for count in freq.values()
        )
        
        return entropy
    
    def detect_exfiltration(self, dns_query):
        """Real-time detection of DNS exfiltration"""
        domain = dns_query['domain']
        entropy = self.calculate_entropy(domain)
        
        # Check multiple indicators
        suspicious_indicators = {
            'high_entropy': entropy > self.entropy_threshold,
            'unusual_length': len(domain) > 50,
            'base64_pattern': self.is_base64_like(domain),
            'high_subdomain_count': domain.count('.') > 4
        }
        
        threat_score = sum(suspicious_indicators.values()) / len(suspicious_indicators)
        
        if threat_score > 0.6:
            return {
                'threat_detected': True,
                'severity': 'HIGH',
                'entropy': round(entropy, 2),
                'indicators': suspicious_indicators,
                'recommended_action': 'BLOCK'
            }

import numpy as np
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import LSTM, Dense, Dropout

class MLDNSDetector:
    def __init__(self):
        self.model = self.build_lstm_model()
        self.model.load_weights('dns_threat_model.h5')
        
    def build_lstm_model(self):
        """LSTM model for sequence analysis of DNS queries"""
        model = Sequential([
            LSTM(128, return_sequences=True, input_shape=(100, 38)),
            Dropout(0.2),
            LSTM(64, return_sequences=True),
            Dropout(0.2),
            LSTM(32),
            Dense(64, activation='relu'),
            Dense(1, activation='sigmoid')  # Binary classification
        ])
        
        model.compile(
            optimizer='adam',
            loss='binary_crossentropy',
            metrics=['accuracy', 'precision', 'recall']
        )
        return model
    
    def analyze_dns_stream(self, dns_stream):
        """Analyze DNS query patterns over time"""
        # Extract features from DNS stream
        features = self.extract_temporal_features(dns_stream)
        
        # Predict probability of exfiltration
        threat_probability = self.model.predict(features)[0][0]
        
        if threat_probability > 0.85:
            # High confidence detection
            self.trigger_incident_response(dns_stream)
            return "CRITICAL: Data exfiltration in progress!"
